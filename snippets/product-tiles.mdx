export const ProductTiles = ({ categories, hideButtons = false }) => {
  const [activeCategory, setActiveCategory] = useState(categories[0].id)
  const [activeSubTab, setActiveSubTab] = useState(
    categories[0].items[0]?.id ?? null
  )
  const [expandedCategories, setExpandedCategories] = useState({})

const currentCategory = categories.find((cat) => cat.id === activeCategory)
const currentSub = currentCategory?.items.find(
(item) => item.id === activeSubTab
)

const toggleCategory = (categoryId) => {
setExpandedCategories(prev => ({
...prev,
[categoryId]: !prev[categoryId]
}))
}

return (

<div className="not-prose space-y-6">
{!hideButtons && (
<div className="flex flex-wrap gap-2">
{categories.map((cat) => (
<button
key={cat.id}
onClick={() => {
setActiveCategory(cat.id)
setActiveSubTab(cat.items[0]?.id ?? null)
}}
className={`px-4 py-2 text-sm rounded-full font-medium transition-colors ${activeCategory === cat.id
              ? 'bg-[#0A0B0D] text-white dark:bg-white dark:text-[#0A0B0D] border-black'
              : 'bg-gray-100 dark:bg-white/10 text-black dark:text-white'
              }`} >
{cat.label}
</button>
))}
</div>
)}

      <div>
        {currentCategory?.items?.length ? (
          <div className="bg-gray-100 dark:bg-[#141414] rounded-xl p-3">
            {(() => {
              const isExpanded = expandedCategories[activeCategory];
              const displayItems = isExpanded ? currentCategory.items : currentCategory.items.slice(0, 30);
              const hasMore = currentCategory.items.length > 20;

              return (
                <>
                  <div className="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-4 xl:grid-cols-6 gap-2">
                    {displayItems.map((item) => (
                      <a
                        key={item.id}
                        href={item.href}
                        onMouseOver={() => setActiveSubTab(item.id)}
                        className={`flex items-center gap-2 text-left px-3 py-2 rounded-lg transition-all ${activeSubTab === item.id
                          ? 'bg-white dark:bg-white/10 shadow-sm'
                          : 'hover:bg-white/50 dark:hover:bg-white/5'
                          }`}
                      >
                        <div className="flex-shrink-0">
                          <img src={item.icon} alt={item.title} className="w-4 h-4" />
                        </div>
                        <div className="flex-1 min-w-0">
                          <h3 className="font-medium text-xs text-black dark:text-white truncate">
                            {item.title}
                          </h3>
                        </div>
                      </a>
                    ))}
                  </div>

                  <div className="mt-6 flex gap-3 justify-center">
                    {hasMore && (
                      <button
                        onClick={() => toggleCategory(activeCategory)}
                        className="inline-flex items-center px-4 py-2 bg-white/50 hover:bg-white/70 dark:bg-white/10 dark:hover:bg-white/20 text-gray-700 dark:text-gray-300 font-medium rounded-lg transition-colors duration-200"
                      >
                        {isExpanded ? 'View Less' : `View More (${currentCategory.items.length - 30} more)`}
                        <svg className={`ml-2 w-4 h-4 transform transition-transform ${isExpanded ? 'rotate-180' : ''}`} fill="none" stroke="currentColor" viewBox="0 0 24 24">
                          <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M19 9l-7 7-7-7" />
                        </svg>
                      </button>
                    )}

                    <a
                      href="/supported-networks"
                      className="inline-flex items-center px-4 py-2 bg-blue-600 hover:bg-blue-700 text-white font-medium rounded-lg transition-colors duration-200"
                    >
                      See All Networks
                      <svg className="ml-2 w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                        <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M9 5l7 7-7 7" />
                      </svg>
                    </a>
                  </div>
                </>
              );
            })()}
          </div>
        ) : (
          <p className="text-sm text-gray-500 dark:text-gray-400">
            No features available for this category.
          </p>
        )}
      </div>
    </div>

)
}
