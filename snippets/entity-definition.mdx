```typescript
import { Entity, PrimaryColumn, Column, Index } from 'typeorm'

@Entity()
export class Transfer {
  @PrimaryColumn()
  id!: string

  @Index()
  @Column('text')
  from!: string

  @Index()
  @Column('text')
  to!: string

  @Column('numeric', { transformer: { to: Number, from: String } })
  value!: bigint

  @Index()
  @Column('text')
  tokenAddress!: string

  @Index()
  @Column('int4')
  blockNumber!: number

  @Column('text')
  transactionHash!: string
}
```
