React.useEffect(() => {
if (!isClient) return;

    const loadNetworks = async () => {
      try {
        setLoading(true);
        setError(null);

        const response = await fetch('https://v2.archive.subsquid.io/usage');
        if (!response.ok) throw new Error('Failed to fetch');

        const data = await response.json();

        const byVM = {};
        const vmCounts = {};

        Object.entries(data).forEach(([slug, network]) => {
          if (network && network.kind) {
            const vm = network.kind.toLowerCase();
            if (!byVM[vm]) {
              byVM[vm] = [];
              vmCounts[vm] = 0;
            }
            byVM[vm].push([slug, network]);
            vmCounts[vm]++;
          }
        });

        // Sort networks by usage
        Object.keys(byVM).forEach(vm => {
          byVM[vm].sort((a, b) => {
            const usageA = a[1]?.usage_bytes || 0;
            const usageB = b[1]?.usage_bytes || 0;
            return usageB - usageA;
          });
        });

        const total = Object.values(vmCounts).reduce((sum, count) => sum + count, 0);

        setNetworksData({ byVM, vmCounts, total });
        setLastUpdated(Date.now());
        setLoading(false);

      } catch (err) {
        console.error('Error loading networks:', err);
        setError('Failed to load networks');
        setLoading(false);
      }
    };

    loadNetworks();

    // Load GitHub data
    const loadGithubData = async () => {
      try {
        const response = await fetch('https://api.github.com/repos/subsquid/squid-sdk');

          stars: data.stargazers_count,
          url: data.html_url
        });

        console.error('Error loading GitHub data:', err);
